39      public JavaMainCallerByFork(AbstractMojo requester1, String mainClassName1, String classpath, String[] jvmArgs1, String[] args1, boolean forceUseArgFile, Toolchain toolchain) throws Exception {
40          super(requester1, mainClassName1, classpath, jvmArgs1, args1);
41          for (String key : System.getenv().keySet()) {
42              env.add(key + "=" + System.getenv(key));
43          }

46          _forceUseArgFile = forceUseArgFile;

49      @Override
50      public boolean run(boolean displayCmd, boolean throwFailure) throws Exception {
51          List<String> cmd = buildCommand();
52          displayCmd(displayCmd, cmd);
53          Executor exec = new DefaultExecutor();
54  
55          //err and out are redirected to out
56          if (!_redirectToLog) {
57            exec.setStreamHandler(new PumpStreamHandler(System.out, System.err, System.in));
58          } else {
59              exec.setStreamHandler(new PumpStreamHandler(new LogOutputStream() {
60                  private LevelState _previous = new LevelState();
61                  
62                  @Override
63                  protected void processLine(String line, int level) {
64                    try {
65                      _previous = LogProcessorUtils.levelStateOf(line, _previous);
66                      switch (_previous.level) {
67                      case ERROR:
68                        requester.getLog().error(line);
69                        break;
70                      case WARNING:
71                        requester.getLog().warn(line);
72                        break;
73                      default:
74                        requester.getLog().info(line);
75                        break;
76                      }
77                    } catch (Exception e) {
78                      e.printStackTrace();
79                    }
80                  }
81              }));
82          }
83  
84          CommandLine cl = new CommandLine(cmd.get(0));
85          for (int i = 1; i < cmd.size(); i++) {
86              cl.addArgument(cmd.get(i), false);
87          }
88          try {
89              int exitValue = exec.execute(cl);
90              if (exitValue != 0) {
91                  if (throwFailure) {
92                      throw new MojoFailureException("command line returned non-zero value:" + exitValue);
93                  }
94                  return false;
95              }
96              if (!displayCmd) tryDeleteArgFile(cmd);
97              return true;
98          } catch (ExecuteException exc) {
99              if (throwFailure) {
100                 throw exc;
101             }
102             return false;
103         }
104     }

106     @Override
107     public SpawnMonitor spawn(boolean displayCmd) throws Exception {
108         List<String> cmd = buildCommand();
109         File out = new File(System.getProperty("java.io.tmpdir"), mainClassName +".out");
110         out.delete();
111         cmd.add(">"+ out.getCanonicalPath());
112         File err = new File(System.getProperty("java.io.tmpdir"), mainClassName +".err");
113         err.delete();
114         cmd.add("2>"+ err.getCanonicalPath());
115         List<String> cmd2 = new ArrayList<String>();
116         if (OS.isFamilyDOS()) {
117             cmd2.add("cmd.exe");
118             cmd2.add("/C");
119             cmd2.addAll(cmd);
120         } else {
121             cmd2.add("/bin/sh");
122             cmd2.add("-c");
123             cmd2.addAll(cmd);
124         }
125         displayCmd(displayCmd, cmd2);
126         ProcessBuilder pb = new ProcessBuilder(cmd2);
127         //pb.redirectErrorStream(true);
128         final Process p = pb.start();
129         return new SpawnMonitor(){
130             @Override
131             public boolean isRunning() throws Exception {
132                 try {
133                     p.exitValue();
134                     return false;
135                 } catch(IllegalThreadStateException e) {
136                     return true;
137                 }
138             }
139         };
140     }

142     private void displayCmd(boolean displayCmd, List<String> cmd) {
143         if (displayCmd) {
144             requester.getLog().info("cmd: " + " " + StringUtils.join(cmd.iterator(), " "));
145         } else if (requester.getLog().isDebugEnabled()) {
146             requester.getLog().debug("cmd: " + " " + StringUtils.join(cmd.iterator(), " "));
147         }
148     }

150     protected List<String> buildCommand() throws Exception {
151         ArrayList<String> back = new ArrayList<String>(2 + jvmArgs.size() + args.size());
152         back.add(_javaExec);
153         if (!_forceUseArgFile && (lengthOf(args, 1) + lengthOf(jvmArgs, 1) < 400)) {
154             back.addAll(jvmArgs);
155             back.add(mainClassName);
156             back.addAll(args);
157         } else {
158             File jarPath = new File(MainHelper.locateJar(MainHelper.class));
159             requester.getLog().debug("plugin jar to add :" + jarPath);
160             addToClasspath(jarPath);
161             back.addAll(jvmArgs);
162             back.add(MainWithArgsInFile.class.getName());
163             back.add(mainClassName);
164             back.add(MainHelper.createArgFile(args).getCanonicalPath());
165         }
166         return back;
167     }

169     private void tryDeleteArgFile(List<String> cmd) throws Exception {
170     	String last = cmd.get(cmd.size() - 1);
171     	if (last.endsWith(MainHelper.argFileSuffix)) {
172     		File f = new File(last);
173     		if (f.exists() && f.getName().startsWith(MainHelper.argFilePrefix)) {
174     			f.delete();
175     		}
176     	}
177     }
178     private long lengthOf(List<String> l, long sepLength) throws Exception {
179         long back = 0;
180         for (String str : l) {
181             back += str.length() + sepLength;
182         }
183         return back;
184     }
