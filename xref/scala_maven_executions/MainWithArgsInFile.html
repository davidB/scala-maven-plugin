7   /**
8    * Use a file and reflection to start a main class with arguments define in a
9    * file. This class should run without other dependencies than jre. This class
10   * is used as a workaround to the windows command line size limitation.
11   *
12   * @author David Bernard
13   */
14  public class MainWithArgsInFile {
15  
16      /**
17       * @param args
18       */
19      public static void main(String[] args) {
20          try {
21              String mainClassName = args[0];
22              List<String> argsFromFile = new ArrayList<>();
23              if (args.length > 0) {
24                  argsFromFile = MainHelper.readArgFile(new File(args[1]));
25              }
26              MainHelper.runMain(mainClassName, argsFromFile, null);
27          } catch (Throwable t) {
28              t.printStackTrace();
29              System.exit(-10000 /* Integer.MIN_VALUE */);
30          }
31      }

33  }
