1   package scala_maven;
2   
3   import org.apache.maven.plugins.annotations.Mojo;
4   import org.apache.maven.plugins.annotations.Parameter;
5   
6   import scala_maven_executions.JavaMainCaller;
7   
8   /**
9    * Display the Scala Compiler help
10   *
11   */
12  @Mojo(name = "help")
13  public class ScalaHelpMojo extends ScalaMojoSupport {
14      /**
15       * Determines if help will only display a version
16       *
17       */
18      @Parameter(property = "maven.scala.help.versionOnly", defaultValue = "false")
19      private boolean versionOnly;
20  
21      @Override
22      public void doExecute() throws Exception {
23          JavaMainCaller jcmd;
24          if (!versionOnly) {
25              jcmd = getScalaCommand();
26              jcmd.addArgs("-help");
27              jcmd.addArgs("-X");
28              jcmd.addArgs("-Y");
29              jcmd.run(displayCmd);
30          }
31          jcmd = getScalaCommand();
32          jcmd.addArgs("-version");
33          jcmd.run(displayCmd);
34      }
35  }
