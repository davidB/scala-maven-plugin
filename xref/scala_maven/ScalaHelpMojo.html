6   /**
7    * Display the Scala Compiler help
8    *
9    * @goal help
10   */
11  public class ScalaHelpMojo extends ScalaMojoSupport {
12      /**
13       * Determines if help will only display a version
14       * @parameter expression="${maven.scala.help.versionOnly}" default-value="false"
15       */
16      private boolean versionOnly;
17  
18      @Override
19      public void doExecute() throws Exception {
20          JavaMainCaller jcmd = null;
21          if (!versionOnly) {
22              jcmd = getScalaCommand();
23              jcmd.addArgs("-help");
24              jcmd.addArgs("-X");
25              jcmd.addArgs("-Y");
26              jcmd.run(displayCmd);
27          }
28          jcmd = getScalaCommand();
29          jcmd.addArgs("-version");
30          jcmd.run(displayCmd);
31      }
32  }
