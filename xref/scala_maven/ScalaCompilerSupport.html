74      @Override
75      protected void doExecute() throws Exception {
76          if (getLog().isDebugEnabled()) {
77              for (File directory : getSourceDirectories()) {
78                  getLog().debug(FileUtils.pathOf(directory, useCanonicalPath));
79              }
80          }
81          File outputDir = FileUtils.fileOf(getOutputDir(), useCanonicalPath);
82          File analysisCacheFile = FileUtils.fileOf(getAnalysisCacheFile(), useCanonicalPath);
83          int nbFiles = compile(getSourceDirectories(), outputDir, analysisCacheFile, getClasspathElements(), false);
84          switch (nbFiles) {
85          case -1:
86              getLog().info("No sources to compile");
87              break;
88          case 0:
89              getLog().info("Nothing to compile - all classes are up to date");
90              break;
91          default:
92              break;
93          }
94      }

96      protected int compile(List<File> sourceRootDirs, File outputDir, File analysisCacheFile,
97          List<String> classpathElements, boolean compileInLoop) throws Exception {
98          if (!compileInLoop && recompileMode == RecompileMode.incremental) {
99              // if not compileInLoop, invoke incrementalCompile immediately
100             long n0 = System.nanoTime();
101             int res = incrementalCompile(classpathElements, sourceRootDirs, outputDir, analysisCacheFile, false);
102             getLog().info(String.format("compile in %.1f s", (System.nanoTime() - n0) / 1_000_000_000.0));
103             return res;
104         }
105 
106         long t0 = System.currentTimeMillis();
107         long n0 = System.nanoTime();
108         LastCompilationInfo lastCompilationInfo = LastCompilationInfo.find(sourceRootDirs, outputDir);
109         if (_lastCompileAt < 0) {
110             _lastCompileAt = lastCompilationInfo.getLastSuccessfullTS();
111         }
112 
113         List<File> files = getFilesToCompile(sourceRootDirs, _lastCompileAt);
114 
115         if (files == null) {
116             return -1;
117         }
118 
119         if (files.size() < 1) {
120             return 0;
121         }
122         if (!outputDir.exists()) {
123             outputDir.mkdirs();
124         }
125         long n1 = System.nanoTime();
126         long t1 = t0 + ((n1 - n0) / 1_000_000);
127 
128         if (compileInLoop && recompileMode == RecompileMode.incremental) {
129             // if compileInLoop, do not invoke incrementalCompile when there's no change
130             int retCode = incrementalCompile(classpathElements, sourceRootDirs, outputDir, analysisCacheFile, true);
131             _lastCompileAt = t1;
132             if (retCode == 1) {
133                 lastCompilationInfo.setLastSuccessfullTS(t1);
134             }
135             return retCode;
136         }
137 
138         getLog().info(
139             String.format("Compiling %d source files to %s at %d", files.size(), outputDir.getAbsolutePath(), t1));
140         JavaMainCaller jcmd = getScalaCommand();
141         jcmd.redirectToLog();
142         if (!classpathElements.isEmpty())
143             jcmd.addArgs("-classpath", MainHelper.toMultiPath(classpathElements));
144         jcmd.addArgs("-d", outputDir.getAbsolutePath());
145         // jcmd.addArgs("-sourcepath", sourceDir.getAbsolutePath());
146         for (File f : files) {
147             jcmd.addArgs(f.getAbsolutePath());
148         }
149         if (jcmd.run(displayCmd, !compileInLoop)) {
150             lastCompilationInfo.setLastSuccessfullTS(t1);
151         } else {
152             compileErrors = true;
153         }
154         getLog().info(String.format("prepare-compile in %.1f s", (n1 - n0) / 1_000_000_000.0));
155         getLog().info(String.format("compile in %.1f s", (System.nanoTime() - n1) / 1_000_000_000.0));
156         _lastCompileAt = t1;
157         return files.size();
158     }

171     private List<File> getFilesToCompile(List<File> sourceRootDirs, long lastSuccessfulCompileTime) throws Exception {
172         List<File> sourceFiles = findSourceWithFilters(sourceRootDirs);
173         if (sourceFiles.size() == 0) {
174             return null;
175         }
176 
177         // filter uptodate
178         // filter is not applied to .java, because scalac failed to used existing .class
179         // for unmodified .java
180         // failed with "error while loading Xxx, class file
181         // '.../target/classes/.../Xxxx.class' is broken"
182         // (restore how it work in 2.11 and failed in 2.12)
183         // TODO a better behavior : if there is at least one .scala to compile then add
184         // all .java, if there is at least one .java then add all .scala (because we
185         // don't manage class dependency)
186         List<File> files = new ArrayList<>(sourceFiles.size());
187         if (_lastCompileAt > 0 || (recompileMode != RecompileMode.all && (lastSuccessfulCompileTime > 0))) {
188             ArrayList<File> modifiedScalaFiles = new ArrayList<>(sourceFiles.size());
189             ArrayList<File> modifiedJavaFiles = new ArrayList<>(sourceFiles.size());
190             for (File f : sourceFiles) {
191                 if (f.lastModified() >= lastSuccessfulCompileTime) {
192                     if (f.getName().endsWith(".java")) {
193                         modifiedJavaFiles.add(f);
194                     } else {
195                         modifiedScalaFiles.add(f);
196                     }
197                 }
198             }
199             if ((modifiedScalaFiles.size() != 0) || (modifiedJavaFiles.size() != 0)) {
200                 files.addAll(sourceFiles);
201                 notifyCompilation(sourceRootDirs);
202             }
203         } else {
204             files.addAll(sourceFiles);
205             notifyCompilation(sourceRootDirs);
206         }
207         return files;
208     }

210     private void notifyCompilation(List<File> files) throws Exception {
211         if (notifyCompilation) {
212             for (File f : files) {
213                 getLog().info(String.format("%s:-1: info: compiling", FileUtils.pathOf(f, useCanonicalPath)));
214             }
215         }
216     }

218     private static class LastCompilationInfo {
219         static LastCompilationInfo find(List<File> sourceRootDirs, File outputDir) {
220             StringBuilder hash = new StringBuilder();
221             for (File f : sourceRootDirs) {
222                 hash.append(f.toString());
223             }
224             return new LastCompilationInfo(
225                 new File(outputDir.getAbsolutePath() + "." + hash.toString().hashCode() + ".timestamp"), outputDir);
226         }
227 
228         private final File _lastCompileAtFile;
229         private final File _outputDir;
230 
231         private LastCompilationInfo(File f, File outputDir) {
232             _lastCompileAtFile = f;
233             _outputDir = outputDir;
234         }
235 
236         long getLastSuccessfullTS() {
237             long back = -1;
238             if (_lastCompileAtFile.exists() && _outputDir.exists() && _outputDir.list().length > 0) {
239                 back = _lastCompileAtFile.lastModified();
240             }
241             return back;
242         }
243 
244         void setLastSuccessfullTS(long v) throws Exception {
245             if (!_lastCompileAtFile.exists()) {
246                 FileUtils.fileWrite(_lastCompileAtFile.getAbsolutePath(), ".");
247             }
248             _lastCompileAtFile.setLastModified(v);
249         }

259     protected File getCompilerBridgeJar() throws Exception {
260         VersionNumber scalaVersion = findScalaVersion();
261         String zincVersion = findVersionFromPluginArtifacts(SBT_GROUP_ID, ZINC_ARTIFACT_ID);
262         return getArtifactJar(SBT_GROUP_ID,
263             scalaVersion.applyScalaArtifactVersioningScheme(COMPILER_BRIDGE_ARTIFACT_ID), zincVersion);

266     private int incrementalCompile(List<String> classpathElements, List<File> sourceRootDirs, File outputDir,
267         File cacheFile, boolean compileInLoop) throws Exception {
268         List<File> sources = findSourceWithFilters(sourceRootDirs);
269         if (sources.isEmpty()) {
270             return -1;
271         }
272 
273         // TODO - Do we really need this duplicated here?
274         if (!outputDir.exists()) {
275             outputDir.mkdirs();
276         }
277 
278         if (incremental == null) {
279             File libraryJar = getLibraryJar();
280             List<File> extraJars = getCompilerDependencies();
281             extraJars.remove(libraryJar);
282             incremental = new SbtIncrementalCompiler(//
283                 libraryJar, //
284                 getReflectJar(), //
285                 getCompilerJar(), //
286                 findScalaVersion(), //
287                 extraJars, //
288                 getCompilerBridgeJar(), //
289                 getLog(), //
290                 cacheFile, //
291                 compileOrder);
292         }
293 
294         classpathElements.remove(outputDir.getAbsolutePath());
295         List<String> scalacOptions = getScalaOptions();
296         List<String> javacOptions = getJavacOptions();
297 
298         try {
299             incremental.compile(classpathElements, sources, outputDir, scalacOptions, javacOptions);
300         } catch (xsbti.CompileFailed e) {
301             if (compileInLoop) {
302                 compileErrors = true;
303             } else {
304                 throw e;
305             }
306         }
307 
308         return 1;
309     }
