88      @Override
89      protected void postCompileActions() throws Exception {
90          if (test == null) {
91              getLog().info("Now running all the unit tests. Use -Dtest=FooTest to run a single test by name");
92          }
93          else {
94              getLog().info("Now running tests matching: " + test);
95          }
96  
97          final InvocationRequest request = new DefaultInvocationRequest();
98          request.setLocalRepositoryDirectory(localRepositoryPath);
99          request.setInteractive(false);
100         request.setErrorHandler(new SystemOutHandler(true));
101         request.setOutputHandler(new SystemOutHandler(true));
102         request.setBaseDirectory(project.getBasedir());
103         request.setPomFile(new File(project.getBasedir(), "pom.xml"));
104 
105         request.setGoals(getMavenGoals());
106         request.setOffline(false);
107 
108         if (test != null) {
109             Properties properties = new Properties();
110             properties.put("test", test);
111             request.setProperties(properties);
112         }
113 
114 
115         if (getLog().isDebugEnabled()) {
116             try {
117                 getLog().debug("Executing: " + new MavenCommandLineBuilder().build(request));
118             }
119             catch (CommandLineConfigurationException e) {
120                 getLog().debug("Failed to display command line: " + e.getMessage());
121             }
122         }
123 
124         try {
125             invoker.execute(request);
126         }
127         catch (final MavenInvocationException e) {
128             getLog().debug("Error invoking Maven: " + e.getMessage(), e);
129             throw new BuildFailureException("Maven invocation failed. " + e.getMessage(), e);
130         }
131     }

133     protected List<String> getMavenGoals() {
