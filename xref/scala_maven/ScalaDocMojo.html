279     @Override
280     protected JavaMainCaller getScalaCommand() throws Exception {
281         //This ensures we have a valid scala version...
282         checkScalaVersion();
283         VersionNumber sv = findScalaVersion();
284         boolean isPreviousScala271 = (new VersionNumber("2.7.1").compareTo(sv) > 0 && !sv.isZero());
285         if (StringUtils.isEmpty(scaladocClassName)) {
286             if (!isPreviousScala271) {
287                 scaladocClassName = "scala.tools.nsc.ScalaDoc";
288             } else {
289                 scaladocClassName = scalaClassName;
290             }
291         }
292 
293         JavaMainCaller jcmd = getEmptyScalaCommand(scaladocClassName);
294         jcmd.addArgs(args);
295         jcmd.addJvmArgs(jvmArgs);
296 
297         if (isPreviousScala271){
298             jcmd.addArgs("-Ydoc");
299         }
300         // copy the classpathElements to not modify the global project definition see https://github.com/davidB/maven-scala-plugin/issues/60
301         List<String> paths = new ArrayList<String>(project.getCompileClasspathElements());
302         paths.remove(project.getBuild().getOutputDirectory()); //remove output to avoid "error for" : error:  XXX is already defined as package XXX ... object XXX {
303         if (!paths.isEmpty())jcmd.addOption("-classpath", MainHelper.toMultiPath(paths));
304         //jcmd.addOption("-sourcepath", sourceDir.getAbsolutePath());
305 
306         boolean isScaladoc2 = (new VersionNumber("2.8.0").compareTo(sv) <= 0 || sv.isZero()) && ("scala.tools.nsc.ScalaDoc".equals(scaladocClassName));
307         if (isScaladoc2) {
308             jcmd.addArgs("-doc-format:html");
309             jcmd.addOption("-doc-title", doctitle);
310         } else {
311             jcmd.addOption("-bottom", getBottomText());
312             jcmd.addOption("-charset", charset);
313             jcmd.addOption("-doctitle", doctitle);
314             jcmd.addOption("-footer", footer);
315             jcmd.addOption("-header", header);
316             jcmd.addOption("-linksource", linksource);
317             jcmd.addOption("-nocomment", nocomment);
318             jcmd.addOption("-stylesheetfile", stylesheetfile);
319             jcmd.addOption("-top", top);
320             jcmd.addOption("-windowtitle", windowtitle);
321         }
322         return jcmd;
323     }

325     @Override
326     public void generate(Sink sink, Locale locale) throws MavenReportException {
327         try {
328             if (!canGenerateReport()) {
329                 getLog().info("No source files found");
330                 return;
331             }
332 
333             File reportOutputDir = getReportOutputDirectory();
334             if (!reportOutputDir.exists()) {
335                 reportOutputDir.mkdirs();
336             }
337             if (StringUtils.isNotEmpty(vscaladocVersion)) {
338                 scaladocClassName = "org.scala_tools.vscaladoc.Main";
339                 BasicArtifact artifact = new BasicArtifact();
340                 artifact.artifactId = "vscaladoc";
341                 artifact.groupId = "org.scala-tools";
342                 artifact.version = vscaladocVersion;
343                 dependencies = new BasicArtifact[]{artifact};
344             }
345 
346             List<File> sources = findSourceFiles();
347             if (sources.size() > 0) {
348                 JavaMainCaller jcmd = getScalaCommand();
349                 jcmd.addOption("-d", reportOutputDir.getAbsolutePath());
350                 for (File x : sources) {
351                     jcmd.addArgs(FileUtils.pathOf(x, useCanonicalPath));
352                 }
353                 jcmd.run(displayCmd);
354             }
355             if (forceAggregate) {
356                 aggregate(project);
357             } else {
358                 // Mojo could not be run from parent after all its children
359                 // So the aggregation will be run after the last child
360                 tryAggregateUpper(project);
361             }
362 
363         } catch (MavenReportException exc) {
364             throw exc;
365         } catch (RuntimeException exc) {
366             throw exc;
367         } catch (Exception exc) {
368             throw new MavenReportException("wrap: " + exc.getMessage(), exc);
369         }
370     }

372     protected void tryAggregateUpper(MavenProject prj) throws Exception {
373         if (prj != null && prj.hasParent() && canAggregate()) {
374             MavenProject parent = prj.getParent();
375             List<MavenProject> modules = parent.getCollectedProjects();
376             if ((modules.size() > 1) && prj.equals(modules.get(modules.size() - 1))) {
377                 aggregate(parent);
378             }
379         }
380     }

382     protected void aggregate(MavenProject parent) throws Exception {
383         List<MavenProject> modules = parent.getCollectedProjects();
384         File dest = new File(parent.getReporting().getOutputDirectory() +"/" + outputDirectory);
385         getLog().info("start aggregation into " + dest);
386         StringBuilder mpath = new StringBuilder();
387         for (MavenProject module : modules) {
388             if ( "pom".equals( module.getPackaging().toLowerCase() ) ) {
389                 continue;
390             }
391             if (aggregateDirectOnly && module.getParent() != parent) {
392                 continue;
393             }
394             File subScaladocPath = new File(module.getReporting().getOutputDirectory() +"/" + outputDirectory).getAbsoluteFile();
395             //System.out.println(" -> " + project.getModulePathAdjustment(module)  +" // " + subScaladocPath + " // " + module.getBasedir() );
396             if (subScaladocPath.exists()) {
397                 mpath.append(subScaladocPath).append(File.pathSeparatorChar);
398             }
399         }
400         if (mpath.length() != 0) {
401             getLog().info("aggregate vscaladoc from : " + mpath);
402             JavaMainCaller jcmd = getScalaCommand();
403             jcmd.addOption("-d", dest.getAbsolutePath());
404             jcmd.addOption("-aggregate", mpath.toString());
405             jcmd.run(displayCmd);
406         } else {
407             getLog().warn("no vscaladoc to aggregate");
408         }
409         tryAggregateUpper(parent);
410     }

412     /**
413      * Method that sets the bottom text that will be displayed on the bottom of
414      * the javadocs.
415      *
416      * @return a String that contains the text that will be displayed at the
417      *         bottom of the javadoc
418      */
419     private String getBottomText() {
420         String inceptionYear = project.getInceptionYear();
421         int actualYear = Calendar.getInstance().get(Calendar.YEAR);
422         String year = String.valueOf(actualYear);
423 
424         String theBottom = StringUtils.replace(bottom, "{currentYear}", year);
425 
426         if (inceptionYear != null) {
427             if (inceptionYear.equals(year)) {
428                 theBottom = StringUtils.replace(theBottom, "{inceptionYear}-", "");
429             } else {
430                 theBottom = StringUtils.replace(theBottom, "{inceptionYear}", inceptionYear);
431             }
432         } else {
433             theBottom = StringUtils.replace(theBottom, "{inceptionYear}-", "");
434         }
435 
436         if (project.getOrganization() == null) {
437             theBottom = StringUtils.replace(theBottom, " {organizationName}", "");
438         } else {
439             if ((project.getOrganization() != null) && (StringUtils.isNotEmpty(project.getOrganization().getName()))) {
440                 if (StringUtils.isNotEmpty(project.getOrganization().getUrl())) {
441                     theBottom = StringUtils.replace(theBottom, "{organizationName}", "<a href=\"" + project.getOrganization().getUrl() + "\">" + project.getOrganization().getName() + "</a>");
442                 } else {
443                     theBottom = StringUtils.replace(theBottom, "{organizationName}", project.getOrganization().getName());
444                 }
445             } else {
446                 theBottom = StringUtils.replace(theBottom, " {organizationName}", "");
447             }
448         }
449 
450         return theBottom;
451     }
