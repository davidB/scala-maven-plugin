104   @Override
105   public void doExecute() throws Exception {
106     if(skip) {
107       getLog().info( "Skipping javadoc generation" );
108       return;
109     }
110     try {
111       generate(null, Locale.getDefault());
112       if(reportOutputDirectory.exists()) {
113         File outputFile = generateArchive( reportOutputDirectory, finalName + "-" + getClassifier() + ".jar" );
114         if(!attach ) {
115           getLog().info( "NOT adding javadoc to attached artifacts list." );
116         } else {
117           // TODO: these introduced dependencies on the project are going to become problematic - can we export it
118           //  through metadata instead?
119           projectHelper.attachArtifact( project, "javadoc", getClassifier(), outputFile );
120         }
121       }
122     } catch (ArchiverException e) {
123       failOnError( "ArchiverException: Error while creating archive", e );
124     } catch (IOException e) {
125       failOnError( "IOException: Error while creating archive", e );
126     } catch ( MavenReportException e ) {
127       failOnError( "MavenReportException: Error while creating archive", e );
128     } catch ( RuntimeException e ) {
129       failOnError( "RuntimeException: Error while creating archive", e );
130     }

135   /**
136    * Method that creates the jar file
137    *
138    * @param javadocFiles the directory where the generated jar file will be put
139    * @param jarFileName the filename of the generated jar file
140    * @return a File object that contains the generated jar file
141    * @throws ArchiverException
142    * @throws IOException 
143    */
144   private File generateArchive( File javadocFiles, String jarFileName ) throws ArchiverException, IOException {
145     final File javadocJar = new File( jarOutputDirectory, jarFileName );
146     if(javadocJar.exists()) {
147       javadocJar.delete();
148     }
149     MavenArchiver archiver = new MavenArchiver();
150     archiver.setArchiver( jarArchiver );
151     archiver.setOutputFile( javadocJar );
152     File contentDirectory = javadocFiles;
153     if(!contentDirectory.exists()) {
154       getLog().warn( "JAR will be empty - no content was marked for inclusion!" );
155     } else {
156       archiver.getArchiver().addDirectory( contentDirectory, DEFAULT_INCLUDES, DEFAULT_EXCLUDES );
157     }
158     List<Resource> resources = project.getBuild().getResources();
159     for ( Resource r : resources ) {
160       if ( r.getDirectory().endsWith( "maven-shared-archive-resources" ) ) {
161         archiver.getArchiver().addDirectory( new File( r.getDirectory() ) );
162       }
163     }
164     if ( useDefaultManifestFile && defaultManifestFile.exists() && archive.getManifestFile() == null ) {
165       getLog().info("Adding existing MANIFEST to archive. Found under: " + defaultManifestFile.getPath());
166       archive.setManifestFile(defaultManifestFile);
167     }
168     try {
169       // we don't want Maven stuff
170       archive.setAddMavenDescriptor( false );
171       archiver.createArchive( project, archive );
172     } catch ( ManifestException e ) {
173       throw new ArchiverException( "ManifestException: " + e.getMessage(), e );
174     } catch ( DependencyResolutionRequiredException e ) {
175       throw new ArchiverException( "DependencyResolutionRequiredException: " + e.getMessage(), e );
176     }
177     return javadocJar;
178   }

184   protected void failOnError(String prefix, Exception e)
185       throws MojoExecutionException {
186     if (failOnError) {

191     }
