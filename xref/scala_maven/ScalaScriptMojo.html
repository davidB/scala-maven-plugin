135     private static String scriptBaseNameOf(File scriptFile, int idx) {
136       if (scriptFile == null) {
137           return "embeddedScript_" + idx;
138       }
139       int dot = scriptFile.getName().lastIndexOf('.');
140       if (dot == -1) {
141           return scriptFile.getName() + "_" + idx;
142       }
143       return scriptFile.getName().substring(0, dot) + "_" + idx;
144     }

146     @Override
147     protected void doExecute() throws Exception {
148         if (script == null && scriptFile == null) {
149             throw new MojoFailureException(
150                     "Either script or scriptFile must be defined");
151         }
152         if (script != null && scriptFile != null) {
153             throw new MojoFailureException(
154                     "Only one of script or scriptFile can be defined");
155         }
156         if (StringUtils.isEmpty(includeScopes)) {
157             if (scriptFile != null) {
158                 includeScopes = "compile, test, runtime";
159             } else {
160                 includeScopes = "plugin";
161             }
162         }
163         if (excludeScopes == null) {
164           excludeScopes = "";
165         }
166         // prepare
167         File scriptDir = new File(outputDir, ".scalaScriptGen");
168         scriptDir.mkdirs();
169         String baseName = scriptBaseNameOf(scriptFile, _lastScriptIndex.incrementAndGet());
170         File destFile = new File(scriptDir, baseName  + ".scala");
171 
172         Set<String> classpath = new HashSet<String>();
173         configureClasspath(classpath);
174 
175 
176         boolean mavenProjectDependency = includeScopes.contains("plugin");
177         wrapScript(destFile, mavenProjectDependency);
178 
179         try {
180             URLClassLoader loader = createScriptClassloader(scriptDir, classpath);
181         	getLog().debug(("classpath : " + Arrays.asList(loader.getURLs())));
182             compileScript(scriptDir, destFile, loader);
183             runScript(mavenProjectDependency, loader, baseName);
184         } finally {
185             if (!keepGeneratedScript) {
186                 delete(scriptDir);
187             }
188         }
189 
190     }

192     private void runScript(boolean mavenProjectDependency, URLClassLoader loader, String baseName) throws Exception {
193         Class<?> compiledScript = loader.loadClass(baseName);
194 
195         ClassLoader currentCL = Thread.currentThread().getContextClassLoader();
196         Thread.currentThread().setContextClassLoader(loader);
197         try {
198             try {
199                 Object instance;
200                 if (mavenProjectDependency) {
201                     Constructor<?> constructor = compiledScript.getConstructor(MavenProject.class, MavenSession.class, Log.class);
202                     instance = constructor.newInstance(project, session, getLog());
203                 } else {
204                     instance = compiledScript.newInstance();
205                 }
206                 try {
207                     compiledScript.getMethod("run").invoke(instance);
208                 } catch (NoSuchMethodException e) {
209                     // ignore because if there is no method then its ok and we
210                     // just
211                     // don't run the method. initialization of the class must
212                     // have been
213                     // enough
214                 }
215             } catch (InvocationTargetException e) {
216                 if (e.getTargetException() != null) {
217                     throw e.getTargetException();
218                 } else if (e.getCause() != null) {
219                     throw e.getCause();
220                 } else {
221                     throw e;
222                 }
223             } catch (ExceptionInInitializerError e) {
224                 if (e.getException() != null) {
225                     throw e.getException();
226                 } else if (e.getCause() != null) {
227                     throw e.getCause();
228                 } else {
229                     throw e;
230                 }
231             }
232         } catch (Throwable e) {
233             if (e instanceof Exception) {
234                 throw (Exception) e;
235             }
236             throw new Exception("A " + e.getClass().getSimpleName() + " exception was thrown", e);
237         } finally {
238             Thread.currentThread().setContextClassLoader(currentCL);
239         }
240     }

242     private URLClassLoader createScriptClassloader(File scriptDir, Set<String> classpath) throws Exception {
243         ClassWorld w = new ClassWorld("zero", null);
244         w.newRealm("mojo", getClass().getClassLoader());
245         Strategy s = new SelfFirstStrategy(w.newRealm("scalaScript", null));
246         ClassRealm rScript = s.getRealm();
247         rScript.setParentClassLoader(getClass().getClassLoader());
248         //rScript.importFrom("mojo", MavenProject.class.getPackage().getName());
249         //rScript.importFrom("mojo", MavenSession.class.getPackage().getName());
250         //rScript.importFrom("mojo", Log.class.getPackage().getName());
251         rScript.importFrom("mojo", "org.apache.maven");
252         // add the script directory to the classpath
253         rScript.addURL(scriptDir.toURI().toURL());
254 
255         for (String string : classpath) {
256         	rScript.addURL(new File(string).toURI().toURL());
257         }
258         return rScript;
259     }

261     private void compileScript(File scriptDir, File destFile, URLClassLoader loader) throws Exception {
262         JavaMainCaller jcmd = getScalaCommand();
263         jcmd.addArgs("-classpath", MainHelper.toClasspathString(loader));
264         jcmd.addArgs("-d", scriptDir.getAbsolutePath());
265         jcmd.addArgs("-sourcepath", scriptDir.getAbsolutePath());
266         jcmd.addArgs(destFile.getAbsolutePath());
267 
268         jcmd.run(displayCmd);
269     }

271     private void configureClasspath(Set<String> classpath) throws Exception, DependencyResolutionRequiredException {
272         Set<String> includes = new TreeSet<String>(Arrays.asList(StringUtils.split(includeScopes.toLowerCase(),",")));
273         Set<String> excludes = new TreeSet<String>(Arrays.asList(StringUtils.split(excludeScopes.toLowerCase(),",")));
274 
275         for(Artifact a : project.getArtifacts()) {
276           if (includes.contains(a.getScope().toLowerCase()) && !excludes.contains(a.getScope())) {
277             addToClasspath(a, classpath, true);
278           }
279         }
280 
281         if (includes.contains("plugin") && !excludes.contains("plugin")) {
282             //Plugin plugin = project.getPlugin("scala-maven-plugin");
283             for(Plugin p : project.getBuildPlugins()) {
284                 if ("scala-maven-plugin".equals(p.getArtifactId())) {
285                     for(Dependency d : p.getDependencies()) {
286                         addToClasspath(factory.createDependencyArtifact(d), classpath, true);
287                     }
288                 }
289             }
290             for(Artifact a : project.getPluginArtifacts()) {
291                 if ("scala-maven-plugin".equals(a.getArtifactId())) {
292                     addToClasspath(a, classpath, true);
293                 }
294             }
295         }
296 
297         if (addToClasspath != null) {
298             classpath.addAll(Arrays.asList(StringUtils.split(addToClasspath,",")));
299         }
300 
301         if (removeFromClasspath != null) {
302             ArrayList<String> toRemove = new ArrayList<String>();
303             String[] jars =  StringUtils.split(removeFromClasspath.trim(),",");
304             for (String string : classpath) {
305                 for (String jar : jars) {
306                     if (string.contains(jar.trim())) {
307                         toRemove.add(string);
308                     }
309                 }
310             }
311             classpath.removeAll(toRemove);
312         }
313 
314 //        String outputDirectory = project.getBuild().getOutputDirectory();
315 //        if(!outputDirectory.endsWith("/")){
316 //            // need it to end with / for URLClassloader
317 //            outputDirectory+="/";
318 //        }
319 //        classpath.add( outputDirectory);
320         addCompilerToClasspath(classpath);
321         addLibraryToClasspath(classpath);
322         //TODO check that every entry from the classpath exists !
323         boolean ok = true;
324         for (String s : classpath) {
325             File f = new File(s);
326             getLog().debug("classpath entry for running and compiling scripts: " + f);
327             if (!f.exists()) {
328                 getLog().error("classpath entry for script not found : " + f);
329                 ok = false;
330             }
331         }
332         if (!ok) {
333             throw new MojoFailureException("some script dependencies not found (see log)");
334         }
335         getLog().debug("Using the following classpath for running and compiling scripts: "+classpath);
336 
337     }

339     private void wrapScript(File destFile, boolean mavenProjectDependency) throws IOException {
340         destFile.delete();
341 
342         FileOutputStream fileOutputStream = new FileOutputStream(destFile);
343         PrintStream out = new PrintStream(fileOutputStream, false, encoding);
344         BufferedReader reader = null;
345         try {
346             if (scriptFile != null) {
347                 reader = new BufferedReader(new InputStreamReader(new FileInputStream(scriptFile), Charset.forName(scriptEncoding)));
348             } else {
349                 reader = new BufferedReader(new StringReader(script));
350             }
351 
352             String baseName = FileUtils.basename(destFile.getName(), ".scala");
353             if (mavenProjectDependency) {
354 //                out.println("import scala.collection.jcl.Conversions._");
355                 out.println("class " + baseName
356                         + "(project :" + MavenProject.class.getCanonicalName()
357                         + ",session :" + MavenSession.class.getCanonicalName()
358                         + ",log :"+Log.class.getCanonicalName()
359                         +") {"
360                         );
361             } else {
362                 out.println("class " + baseName + " {");
363             }
364 
365             String line = reader.readLine();
366             while (line != null) {
367                 out.print("  ");
368                 out.println(line);
369                 line = reader.readLine();
370             }
371 
372             out.println("}");
373         } finally {
374             IOUtil.close(out);
375             IOUtil.close(fileOutputStream);
376             IOUtil.close(reader);
377         }
378     }

380     private void delete(File scriptDir) {
381         if (scriptDir.isDirectory()) {
382             for (File file : scriptDir.listFiles()) {
383                 delete(file);
384             }
385         }
386 
387         scriptDir.deleteOnExit();
388         scriptDir.delete();
