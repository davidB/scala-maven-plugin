64      protected void initFilters() throws Exception {
65        if (includes.isEmpty()) {
66          includes.add("**/*.scala");
67          if (sendJavaToScalac && isJavaSupportedByCompiler()) {
68              includes.add("**/*.java");
69          }
70        }
71        if (!_filterPrinted && getLog().isInfoEnabled()) {
72          StringBuilder builder = new StringBuilder("includes = [");
73          for (String include : includes) {
74              builder.append(include).append(",");
75          }
76          builder.append("]");
77          getLog().info(builder.toString());
78  
79          builder = new StringBuilder("excludes = [");
80          for (String exclude : excludes) {
81              builder.append(exclude).append(",");
82          }
83          builder.append("]");
84          getLog().info(builder.toString());
85          _filterPrinted = true;
86        }
87      }
88      
89      /**
90       * Finds all source files in a set of directories with a given extension.
91       */
92      protected List<File> findSourceWithFilters(List<File> sourceRootDirs) throws Exception {
93          List<File> sourceFiles = new ArrayList<File>();
94          
95          initFilters();
96  
97          // TODO - Since we're making files anyway, perhaps we should just test
98          // for existence here...
99          for (File dir : sourceRootDirs) {
100             String[] tmpFiles = MainHelper.findFiles(dir, includes.toArray(new String[includes.size()]), excludes.toArray(new String[excludes.size()]));
101             for (String tmpLocalFile : tmpFiles) {
102                 File tmpAbsFile = FileUtils.fileOf(new File(dir, tmpLocalFile), useCanonicalPath);
103                 sourceFiles.add(tmpAbsFile);
104             }
105         }
106         //scalac is sensible to scala file order, file system can't garanty file order => unreproductible build error across platform
107         // to garanty reproductible command line we order file by path (os dependend).
108         Collections.sort(sourceFiles);
109         return sourceFiles;
110     }

112     /**
113      * This limits the source directories to only those that exist for real.
114      */
115     protected List<File> normalize(List<String> compileSourceRootsList) throws Exception {
116         List<File> newCompileSourceRootsList = new ArrayList<File>();
117         if (compileSourceRootsList != null) {
118             // copy as I may be modifying it
119             for (String srcDir : compileSourceRootsList) {
120                 File srcDirFile = FileUtils.fileOf(new File(srcDir), useCanonicalPath);
121                 if (!newCompileSourceRootsList.contains(srcDirFile) && srcDirFile.exists()) {
122                     newCompileSourceRootsList.add(srcDirFile);
123                 }
124             }
125         }
126         return newCompileSourceRootsList;
127     }
