4   import java.util.ArrayList;
5   import java.util.Collections;

10  import org.apache.maven.plugins.annotations.Parameter;
11  
12  import scala_maven_executions.MainHelper;
13  
14  /**
15   * @author david.bernard
16   */
17  abstract public class ScalaSourceMojoSupport extends ScalaMojoSupport {
18  
19      /**
20       * Enables/Disables sending java source to the scala compiler.
21       *
22       */
23      @Parameter(defaultValue="true")
24      protected boolean sendJavaToScalac = true;
25  
26      /**
27       * A list of inclusion filters for the compiler.
28       * ex :
29       * <pre>
30       *    &lt;includes&gt;
31       *      &lt;include&gt;SomeFile.scala&lt;/include&gt;
32       *    &lt;/includes&gt;
33       * </pre>
34       *
35       */
36      @Parameter
37      protected Set<String> includes = new HashSet<String>();
38  
39      /**
40       * A list of exclusion filters for the compiler.
41       * ex :
42       * <pre>
43       *    &lt;excludes&gt;
44       *      &lt;exclude&gt;SomeBadFile.scala&lt;/exclude&gt;
45       *    &lt;/excludes&gt;
46       * </pre>
47       *
48       */
49      @Parameter
50      protected Set<String> excludes = new HashSet<String>();
51  
52      /**
53       * Retrieves the list of *all* root source directories.  We need to pass all .java and .scala files into the scala compiler
54       */
55      abstract protected List<File> getSourceDirectories() throws Exception;
56  
57      private boolean _filterPrinted = false;
58  
59      /**
60       * Finds all source files in a set of directories with a given extension.
61       */
62      protected List<File> findSourceWithFilters() throws Exception {
63          return  findSourceWithFilters(getSourceDirectories());
64      }
65  
66      protected void initFilters() throws Exception {
67        if (includes.isEmpty()) {
68          includes.add("**/*.scala");
69          if (sendJavaToScalac && isJavaSupportedByCompiler()) {
70              includes.add("**/*.java");
71          }
72        }
73        if (!_filterPrinted && getLog().isDebugEnabled()) {
74          StringBuilder builder = new StringBuilder("includes = [");
75          for (String include : includes) {
76              builder.append(include).append(",");
77          }
78          builder.append("]");
79          getLog().debug(builder.toString());
80  
81          builder = new StringBuilder("excludes = [");
82          for (String exclude : excludes) {
83              builder.append(exclude).append(",");
84          }
85          builder.append("]");
86          getLog().debug(builder.toString());
87          _filterPrinted = true;
88        }
89      }
90  
91      /**
92       * Finds all source files in a set of directories with a given extension.
93       */
94      protected List<File> findSourceWithFilters(List<File> sourceRootDirs) throws Exception {
95          List<File> sourceFiles = new ArrayList<File>();
96  
97          initFilters();
98  
99          // TODO - Since we're making files anyway, perhaps we should just test
100         // for existence here...
101         for (File dir : sourceRootDirs) {
102             String[] tmpFiles = MainHelper.findFiles(dir, includes.toArray(new String[includes.size()]), excludes.toArray(new String[excludes.size()]));
103             for (String tmpLocalFile : tmpFiles) {
104                 File tmpAbsFile = FileUtils.fileOf(new File(dir, tmpLocalFile), useCanonicalPath);
105                 sourceFiles.add(tmpAbsFile);
106             }
107         }
108         // scalac is sensitive to scala file order, file system can't guarantee file order => unreproducible build error across platforms
109         // sort files by path (OS dependent) to guarantee reproducible command line.
110         Collections.sort(sourceFiles);
111         return sourceFiles;
112     }
113 
114     /**
115      * This limits the source directories to only those that exist for real.
116      */
117     protected List<File> normalize(List<String> compileSourceRootsList) throws Exception {
118         List<File> newCompileSourceRootsList = new ArrayList<File>();
119         if (compileSourceRootsList != null) {
120             // copy as I may be modifying it
121             for (String srcDir : compileSourceRootsList) {
122                 File srcDirFile = FileUtils.fileOf(new File(srcDir), useCanonicalPath);
123                 if (!newCompileSourceRootsList.contains(srcDirFile) && srcDirFile.exists()) {
124                     newCompileSourceRootsList.add(srcDirFile);
125                 }
126             }
127         }
128         return newCompileSourceRootsList;
129     }
