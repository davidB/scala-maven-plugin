557     @Override
558     public void execute() throws MojoExecutionException, MojoFailureException {
559         try {
560             String oldWay = System.getProperty("maven.scala.version");
561             if (oldWay != null) {
562                 getLog().warn("using 'maven.scala.version' is deprecated, use 'scala.version' instead");
563                 if (scalaVersion != null) {
564                     scalaVersion = oldWay;
565                 }
566             }
567 
568             oldWay = System.getProperty("maven.scala.displayCmd");
569             if (oldWay != null) {
570                 getLog().warn("using 'maven.scala.displayCmd' is deprecated, use 'displayCmd' instead");
571                 displayCmd = displayCmd || Boolean.parseBoolean(oldWay);
572             }
573             checkScalaVersion();
574             doExecute();
575         } catch (MojoExecutionException exc) {
576             throw exc;
577         } catch (MojoFailureException exc) {
578             throw exc;
579         } catch (RuntimeException exc) {
580             throw exc;
581         } catch (Exception exc) {
582             throw new MojoExecutionException("wrap: " + exc, exc);
583         }
584     }

590     protected VersionNumber findScalaVersion() throws Exception {
591         if (_scalaVersionN == null) {
592             String detectedScalaVersion = scalaVersion;
593             if (StringUtils.isEmpty(detectedScalaVersion)) {
594                 detectedScalaVersion = findScalaVersionFromDependencies();
595             }
596             if (StringUtils.isEmpty(detectedScalaVersion)) {
597                 if (!ScalaMojoSupport.POM.equals( project.getPackaging().toLowerCase() )) {
598                     getLog().warn("you don't define "+ getScalaOrganization() + ":" + SCALA_LIBRARY_ARTIFACTID + " as a dependency of the project");
599                 }
600                 detectedScalaVersion = "0.0.0";
601             } else {
602                 // grappy hack to retrieve the SNAPSHOT version without timestamp,...
603                 // because if version is -SNAPSHOT and artifact is deploy with uniqueValue then the version
604                 // get from dependency is with the timestamp and a build number (the resolved version)
605                 // but scala-compiler with the same version could have different resolved version (timestamp,...)
606                 boolean isSnapshot = ArtifactUtils.isSnapshot(detectedScalaVersion);
607                 if (isSnapshot && !detectedScalaVersion.endsWith("-SNAPSHOT")) {
608                     detectedScalaVersion = detectedScalaVersion.substring(0, detectedScalaVersion.lastIndexOf('-', detectedScalaVersion.lastIndexOf('-')-1)) + "-SNAPSHOT";
609                 }
610             }
611             if (StringUtils.isEmpty(detectedScalaVersion)) {
612                 throw new MojoFailureException("no scalaVersion detected or set");
613             }
614             if (StringUtils.isNotEmpty(scalaVersion)) {
615                 if (!scalaVersion.equals(detectedScalaVersion)) {
616                     getLog().warn("scala library version define in dependencies doesn't match the scalaVersion of the plugin");
617                 }
618                 //getLog().info("suggestion: remove the scalaVersion from pom.xml"); //scalaVersion could be define in a parent pom where lib is not required
619             }
620             _scalaVersionN = new VersionNumber(detectedScalaVersion);
621         }
622         return _scalaVersionN;
623     }

625     private String findScalaVersionFromDependencies() throws Exception {

629     //TODO refactor to do only one scan of dependencies to find version
630     protected String findVersionFromDependencies(String groupId, String artifactId) throws Exception {
631         String version = null;
632         for (Dependency dep : getDependencies()) {
633             if (groupId.equals(dep.getGroupId()) && artifactId.equals(dep.getArtifactId())) {
634                 version = dep.getVersion();
635             }
636         }
637         if (StringUtils.isEmpty(version)) {
638             List<Dependency> deps = new ArrayList<Dependency>();
639             deps.addAll(project.getModel().getDependencies());
640             if (project.getModel().getDependencyManagement() != null) {
641                 deps.addAll(project.getModel().getDependencyManagement().getDependencies());
642             }
643             for (Dependency dep : deps) {
644                 if (groupId.equals(dep.getGroupId()) && artifactId.equals(dep.getArtifactId())) {
645                     version = dep.getVersion();
646                 }
647             }
648         }
649         return version;
650     }

652     protected void checkScalaVersion() throws Exception {
653         String sv = findScalaVersion().toString();
654         if (StringUtils.isNotEmpty(scalaHome)) {
655             getLog().warn(String.format("local scala-library.jar and scala-compiler.jar from scalaHome(%s) used instead of scala %s", scalaHome, sv));
656         }
657         if (checkMultipleScalaVersions) {
658             checkCorrectVersionsOfScalaLibrary(sv);
659         }
660     }

662     /** this method checks to see if there are multiple versions of the scala library
663      * @throws Exception */
664     private void checkCorrectVersionsOfScalaLibrary(String scalaDefVersion) throws Exception {
665         getLog().debug("Checking for multiple versions of scala");
666         //TODO - Make sure we handle bad artifacts....
667         // TODO: note that filter does not get applied due to MNG-3236
668         VersionNumbermber">VersionNumber sv = new VersionNumber(scalaDefVersion);
669         VersionNumber requiredScalaVersion = StringUtils.isNotEmpty(scalaCompatVersion) ? new VersionNumberMask(scalaCompatVersion) : sv;
670         if (requiredScalaVersion.compareTo(sv) != 0) {
671           String msg = String.format("Scala library detected %s doesn't match scala.compat.version : %s", sv, requiredScalaVersion);
672           if(failOnMultipleScalaVersions) {

674             throw new MojoFailureException(msg);

676           getLog().warn(msg);

679         request.setProject(project);
680         checkArtifactForScalaVersion(requiredScalaVersion, dependencyTreeBuilder.buildDependencyGraph(request, null));
681     }

685     private void checkArtifactForScalaVersion(VersionNumber requiredScalaVersion, DependencyNode rootNode) throws Exception {
686         final CheckScalaVersionVisitorckScalaVersionVisitor">CheckScalaVersionVisitor visitor = new CheckScalaVersionVisitor(requiredScalaVersion, getLog(), getScalaOrganization());
687 
688         CollectingDependencyNodeVisitor collectingVisitor = new CollectingDependencyNodeVisitor();
689         DependencyNodeVisitor firstPassVisitor = new FilteringDependencyNodeVisitor( collectingVisitor, createScalaDistroDependencyFilter() );
690         rootNode.accept( firstPassVisitor );
691 
692         DependencyNodeFilter secondPassFilter = new AncestorOrSelfDependencyNodeFilter( collectingVisitor.getNodes() );
693         DependencyNodeVisitor filteredVisitor = new FilteringDependencyNodeVisitor( visitor, secondPassFilter );
694 
695         rootNode.accept( filteredVisitor );
696 
697         if(visitor.isFailed()) {
698             visitor.logScalaDependents();
699             if(failOnMultipleScalaVersions) {
700                 getLog().error("Multiple versions of scala libraries detected!");
701                 throw new MojoFailureException("Multiple versions of scala libraries detected!");
702             }
703             getLog().warn("Multiple versions of scala libraries detected!");
704         }
705     }

727     /**
728      * Get a {@link JavaMainCaller} used invoke a Java process. Typically this
729      * will be one of the Scala utilities (Compiler, ScalaDoc, REPL, etc.).
730      * <p>
731      * This method does some setup on the {@link JavaMainCaller} which is not
732      * done by merely invoking {@code new} on one of the
733      * implementations. Specifically, it adds any Scala compiler plugin options,
734      * JVM options, and Scalac options defined on the plugin.
735      *
736      * @param forkOverride override the setting for {@link #fork}. Currently
737      *        this should only be set if you are invoking the REPL.
738      * @param mainClass the JVM main class to invoke.
739      *
740      * @return a {@link JavaMainCaller} to use to invoke the given command.
741      */
742     protected final JavaMainCaller getScalaCommand(final boolean forkOverride,
743                                                    final String mainClass) throws Exception {
744         JavaMainCaller cmd = getEmptyScalaCommand(mainClass, forkOverride);
745         cmd.addArgs(args);
746         if (StringUtils.isNotEmpty(addScalacArgs)) {
747           cmd.addArgs(StringUtils.split(addScalacArgs, "|"));
748         }
749         addCompilerPluginOptions(cmd);
750         cmd.addJvmArgs(jvmArgs);
751         return cmd;
752     }

766     /**
767      * Get a {@link JavaMainCaller} used invoke a Java process. Typically this
768      * will be one of the Scala utilities (Compiler, ScalaDoc, REPL, etc.).
769      *
770      * @param mainClass the JVM main class to invoke.
771      * @param forkOverride override the setting for {@link #fork}. Currently
772      *        this should only be set if you are invoking the REPL.
773      *
774      * @return a {@link JavaMainCaller} to use to invoke the given command.
775      */
776     protected JavaMainCaller getEmptyScalaCommand(final String mainClass,
777                                                   final boolean forkOverride) throws Exception {
778 
779         // If we are deviating from the plugin settings, let the user know
780         // what's going on.
781         if (forkOverride != this.fork) {
782             super.getLog().info("Fork behavior overridden");
783             super.getLog().info(String.format("Fork for this execution is %s.", String.valueOf(forkOverride)));
784         }
785 
786         //TODO - Fork or not depending on configuration?
787         JavaMainCaller cmd;
788         String toolcp = getToolClasspath();
789         if(forkOverride) {
790             // HACK (better may need refactor)
791             boolean bootcp = true;
792             if (args != null) {
793                 for(String arg : args) {
794                     bootcp = bootcp && !"-nobootcp".equals(arg);
795                 }
796             }
797             String cp = bootcp ? "" : toolcp;
798             bootcp = bootcp && !(StringUtils.isNotEmpty(addScalacArgs) && addScalacArgs.contains("-nobootcp"));
799             // scalac with args in files
800             // * works only since 2.8.0
801             // * is buggy (don't manage space in path on windows)
802             getLog().debug("use java command with args in file forced : " + forceUseArgFile);
803             cmd = new JavaMainCallerByFork(this, mainClass, cp, null, null, forceUseArgFile, toolchainManager.getToolchainFromBuildContext("jdk", session));
804             if (bootcp) {
805                 cmd.addJvmArgs("-Xbootclasspath/a:" + toolcp);
806             }
807         } else  {
808             cmd = new JavaMainCallerInProcess(this, mainClass, toolcp, null, null);
809         }
810         return cmd;
811     }

879     protected List<File> getCompilerDependencies() throws Exception {
880       List<File> d = new ArrayList<File>();
881       if(StringUtils.isEmpty(scalaHome)) {
882         for (Artifact artifact : getAllDependencies(getScalaOrganization(), SCALA_COMPILER_ARTIFACTID, findScalaVersion().toString())) {
883           d.add(artifact.getFile());
884         }
885       } else {
886         for(File f : new File(scalaHome, "lib").listFiles()) {
887           String name = f.getName();
888           if (name.endsWith(".jar") && !name.contains("scala-library") && !name.contains("scala-compiler")) {
889             d.add(f);
890           }
891         }
892       }
893       return d;
894     }

949      * Retrieves a list of paths to scala compiler plugins.
950      *
951      * @return The list of plugins
952      * @throws Exception
953      */
954     private Set<String> getCompilerPlugins() throws Exception {
955         Set<String> plugins = new HashSet<String>();
956         if (compilerPlugins != null) {
957             Set<String> ignoreClasspath = new LinkedHashSet<String>();
958             addCompilerToClasspath(ignoreClasspath);
959             addLibraryToClasspath(ignoreClasspath);
960             for (BasicArtifact artifact : compilerPlugins) {
961                 getLog().info("compiler plugin: " + artifact.toString());
962                 // TODO - Ensure proper scala version for plugins
963                 Set<String> pluginClassPath = new HashSet<String>();
964                 //TODO - Pull in transitive dependencies.
965                 addToClasspath(artifact.groupId, artifact.artifactId, artifact.version, artifact.classifier, pluginClassPath, false);
966                 pluginClassPath.removeAll(ignoreClasspath);
967                 plugins.addAll(pluginClassPath);
968             }
969         }
970         return plugins;
971     }
